2100MERROR_RETVAL retroflat_loop_generic(
2101 retroflat_loop_iter frame_iter, retroflat_loop_iter loop_iter, void* data
2102) {
2103 MERROR_RETVAL retval = MERROR_OK;
2104 retroflat_ms_t next = 0,
2105 now = 0;
2106
2107 g_retroflat_state->loop_iter = (retroflat_loop_iter)loop_iter;
2108 g_retroflat_state->loop_data = (void*)data;
2109 g_retroflat_state->frame_iter = (retroflat_loop_iter)frame_iter;
2110
2111 if(
2112 RETROFLAT_FLAGS_RUNNING ==
2113 (g_retroflat_state->retroflat_flags & RETROFLAT_FLAGS_RUNNING)
2114 ) {
2115 /* Main loop is already running, so we're just changing the iter call
2116 * and leaving!
2117 */
2118 debug_printf( 1, "main loop already running!" );
2119 goto cleanup;
2120 }
2121
2122 g_retroflat_state->retroflat_flags |= RETROFLAT_FLAGS_RUNNING;
2123 do {
2124 if(
2125 /* Not waiting for the next frame? */
2126 RETROFLAT_FLAGS_WAIT_FOR_FPS !=
2127 (RETROFLAT_FLAGS_WAIT_FOR_FPS & g_retroflat_state->retroflat_flags) &&
2128 /* Inter-frame loop present? */
2129 NULL != g_retroflat_state->loop_iter
2130 ) {
2131 /* Run the loop iter as many times as possible. */
2132 g_retroflat_state->loop_iter( g_retroflat_state->loop_data );
2133 }
2134 if(
2135 RETROFLAT_FLAGS_UNLOCK_FPS !=
2136 (RETROFLAT_FLAGS_UNLOCK_FPS & g_retroflat_state->retroflat_flags) &&
2137 retroflat_get_ms() < next
2138 ) {
2139 /* Sleep/low power for a bit. */
2140 continue;
2141 }
2142
2143 retroflat_heartbeat_update();
2144
2145 if( NULL != g_retroflat_state->frame_iter ) {
2146 /* Run the frame iterator once per FPS tick. */
2147 g_retroflat_state->frame_iter( g_retroflat_state->loop_data );
2148 }
2149 /* Reset wait-for-frame flag AFTER frame callback. */
2150 g_retroflat_state->retroflat_flags &= ~RETROFLAT_FLAGS_WAIT_FOR_FPS;
2151 now = retroflat_get_ms();
2152 if( now + retroflat_fps_next() > now ) {
2153 next = now + retroflat_fps_next();
2154 } else {
2155 /* Rollover protection. */
2156 /* TODO: Add difference from now/next to 0 here. */
2157 next = 0;
2158 }
2159 } while(
2160 RETROFLAT_FLAGS_RUNNING ==
2161 (RETROFLAT_FLAGS_RUNNING & g_retroflat_state->retroflat_flags)
2162 );
2163 retval = g_retroflat_state->retval;
2164
2165cleanup:
2166
2167 /* This should be set by retroflat_quit(). */
2168 return retval;
2169}
