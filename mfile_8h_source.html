261off_t mfile_file_has_bytes( struct MFILE_CADDY* p_file ) {
262 size_t cursor = 0;
263 cursor = p_file->cursor( p_file );
264 if( 0 <= cursor ) {
265#if MFILE_READ_TRACE_LVL > 0
266 debug_printf( MFILE_READ_TRACE_LVL,
267 "file has " OFF_T_FMT " bytes left...",
268 p_file->sz - cursor );
269#endif /* MFILE_READ_TRACE_LVL */
270 return p_file->sz - cursor;
271 } else {
272#if MFILE_READ_TRACE_LVL > 0
273 /* TODO: Improved error message/handling. */
274 debug_printf( MFILE_READ_TRACE_LVL, "file has error bytes left!" );
275#endif /* MFILE_READ_TRACE_LVL */
276 return 0;
277 }

292 if( NULL != ext_ptr ) {
293 *ext_ptr = '\0';

302MERROR_RETVAL mfile_file_read_int(
303 struct MFILE_CADDY* p_file, uint8_t* buf, size_t buf_sz, uint8_t flags
304) {
305 MERROR_RETVAL retval = MERROR_OK;
306
307 if(
308#ifdef MAUG_LSBF
309 MFILE_READ_FLAG_LSBF == (MFILE_READ_FLAG_LSBF & flags)
310#elif defined( MAUG_MSBF )
311 MFILE_READ_FLAG_MSBF != (MFILE_READ_FLAG_MSBF & flags)
312#endif
313 ) {
314 debug_printf( MFILE_READ_TRACE_LVL, "reading integer forward" );
315 /* Shrink the buffer moving right and read into it. */
316 retval = p_file->read_block( p_file, buf, buf_sz );
317
318 } else {
319 debug_printf( MFILE_READ_TRACE_LVL, "reading integer reversed" );
320 /* Move to the end of the output buffer and read backwards. */
321 while( 0 < buf_sz ) {
322 retval = p_file->read_byte( p_file, (buf + (buf_sz - 1)) );
323 maug_cleanup_if_not_ok();
324 buf_sz--;
325 }
326 }
327
328cleanup:
329
330 return retval;
331}

368MERROR_RETVAL mfile_mem_read_block(
369 struct MFILE_CADDY* p_file, uint8_t* buf, size_t buf_sz
370) {
371
372 if( p_file->mem_cursor >= p_file->sz ) {
373 return MERROR_FILE;
374 }
375
376 while( 0 < buf_sz-- ) {
377 *(buf++) = p_file->mem_buffer[p_file->mem_cursor++];
378 }
379
380 return MERROR_OK;
381}

401MERROR_RETVAL mfile_mem_read_line(
402 struct MFILE_CADDY* p_f, char* buffer, off_t buffer_sz, uint8_t flags
403) {
404 MERROR_RETVAL retval = MERROR_OK;
405 off_t i = 0;
406
407 assert( MFILE_CADDY_TYPE_MEM_BUFFER == p_f->type );
408
409 while( i < buffer_sz - 1 && p_f->has_bytes( p_f ) ) {
410 /* Check for potential overflow. */
411 if( i + 1 >= buffer_sz ) {
412 error_printf( "overflow reading string from file!" );
413 retval = MERROR_FILE;
414 break;
415 }
416
417 p_f->read_int( p_f, (uint8_t*)&(buffer[i]), 1, 0 );
418 if( '\n' == buffer[i] ) {
419 /* Break on newline and overwrite it below. */
420 break;
421 }
422 i++;
423 }
424
425 assert( i < buffer_sz ); /* while() above stops before buffer_sz! */
426
427 /* Append a null terminator. */
428 buffer[i] = '\0';
429
430 return retval;
431}

478 MERROR_RETVAL retval = MERROR_OK;
479
480 debug_printf( 1,
481 "locking handle %p as file %p (" OFF_T_FMT " bytes)...",

494 p_file->read_line = mfile_mem_read_line;
495 p_file->write_block = mfile_mem_write_block;
496
497 p_file->sz = handle_sz;

525void mfile_close( mfile_t* p_file ) {
526# if MFILE_SEEK_TRACE_LVL > 0
527 debug_printf( MFILE_SEEK_TRACE_LVL, "closing file..." );
528# endif /* MFILE_SEEK_TRACE_LVL */
529# ifdef MFILE_MMAP
530 munmap( bytes_ptr_h, bytes_sz );
531# else
532 /* maug_mfree( bytes_ptr_h ); */
533 switch( p_file->type ) {
534 case 0:
535 /* Do nothing silently. */
536 break;
537
538 case MFILE_CADDY_TYPE_FILE:
539 mfile_plt_close( p_file );
540 p_file->type = 0;
541 break;
542
543 case MFILE_CADDY_TYPE_MEM_BUFFER:
544 if( NULL != p_file->mem_buffer ) {
545 maug_munlock( p_file->h.mem, p_file->mem_buffer );
546 debug_printf( 1, "unlocked handle %p from file %p...",
547 p_file->h.mem, p_file );
548 p_file->type = 0;
549 }
550 break;
551
552 default:
553 error_printf( "unknown file type: %d", (p_file)->type );
554 break;
555 }
556# endif /* MFILE_MMAP */
557}
