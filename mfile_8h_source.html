127MERROR_RETVAL mfile_mem_read_int(
128 struct MFILE_CADDY* p_file, uint8_t* buf, size_t buf_sz, uint8_t flags
129) {
130 MERROR_RETVAL retval = MERROR_OK;
131
132 assert( MFILE_CADDY_TYPE_MEM_BUFFER == p_file->type );
133
134 if( MFILE_READ_FLAG_LSBF != (MFILE_READ_FLAG_LSBF & flags) ) {
135 /* Shrink the buffer moving right and read into it. */
136 while( 0 < buf_sz ) {
137 /* Check for EOF. */
138 if( p_file->mem_cursor >= p_file->sz ) {
139 retval = MERROR_FILE;
140 error_printf(
141 "cursor " OFF_T_FMT " beyond end of buffer " OFF_T_FMT "!",
142 p_file->mem_cursor, p_file->sz );
143 goto cleanup;
144 }
145
146 buf[buf_sz - 1] = p_file->mem_buffer[p_file->mem_cursor];
147 debug_printf( MFILE_TRACE_LVL, "byte #" SIZE_T_FMT " = # " OFF_T_FMT,
148 buf_sz - 1, p_file->mem_cursor );
149 buf_sz--;
150 p_file->mem_cursor++;
151 }
152
153 } else {
154 /* Move to the end of the output buffer and read backwards. */
155 while( 0 < buf_sz ) {
156 /* Check for EOF. */
157 if( p_file->mem_cursor >= p_file->sz ) {
158 retval = MERROR_FILE;
159 error_printf(
160 "cursor " OFF_T_FMT " beyond end of buffer " OFF_T_FMT "!",
161 p_file->mem_cursor, p_file->sz );
162 goto cleanup;
163 }
164
165 buf[buf_sz - 1] = p_file->mem_buffer[p_file->mem_cursor];
166 debug_printf( MFILE_TRACE_LVL, "byte #" SIZE_T_FMT " = # " OFF_T_FMT,
167 buf_sz - 1, p_file->mem_cursor );
168 buf_sz--;
169 buf++;
170 p_file->mem_cursor++;
171 }
172 }
173
174cleanup:
175
176 return retval;
177}

197MERROR_RETVAL mfile_mem_read_line(
198 struct MFILE_CADDY* p_f, char* buffer, off_t buffer_sz, uint8_t flags
199) {
200 MERROR_RETVAL retval = MERROR_OK;
201 off_t i = 0;
202
203 assert( MFILE_CADDY_TYPE_MEM_BUFFER == p_f->type );
204
205 while( i < buffer_sz - 1 && mfile_has_bytes( p_f ) ) {
206 /* Check for potential overflow. */
207 if( i + 1 >= buffer_sz ) {
208 error_printf( "overflow reading string from file!" );
209 retval = MERROR_FILE;
210 break;
211 }
212
213 p_f->read_int( p_f, (uint8_t*)&(buffer[i]), 1, 0 );
214 if( '\n' == buffer[i] ) {
215 /* Break on newline and overwrite it below. */
216 break;
217 }
218 i++;
219 }
220
221 assert( i < buffer_sz ); /* while() above stops before buffer_sz! */
222
223 /* Append a null terminator. */
224 buffer[i] = '\0';
225
226 return retval;
227}

247

251}

258 size_t i = 0;

264# endif /* MFILE_MMAP */

266 /* MAUG_MHANDLE* p_bytes_ptr_h, off_t* p_bytes_sz */
267
268 /* TODO: Move MVFS stuff into its own platform handler or something? */
269# if defined( MVFS_ENABLED )
270
271 while( NULL != gc_mvfs_data[i] ) {
272 if( 0 == strcmp( filename, gc_mvfs_filenames[i] ) ) {
273 debug_printf( 1, "found file \"%s\" at VFS index: " SIZE_T_FMT
274 " (size: " OFF_T_FMT " bytes)",
275 filename, i, *(gc_mvfs_lens[i]) );
276 break;
277 }
278 i++;
279 }
280
281 if( NULL == gc_mvfs_data[i] ) {
282 retval = MERROR_FILE;
283 error_printf( "file \"%s\" not found in VFS!", filename );
284 goto cleanup;
285 }
286
287 p_file->type = MFILE_CADDY_TYPE_MEM_BUFFER;
288 p_file->read_int = mfile_mem_read_int;
289 p_file->seek = mfile_mem_seek;
290 p_file->read_line = mfile_mem_read_line;
291 p_file->flags = MFILE_FLAG_READ_ONLY;
292 p_file->mem_buffer = gc_mvfs_data[i];
293 p_file->sz = *(gc_mvfs_lens[i]);
294 p_file->mem_cursor = 0;
295
296cleanup:
297
298# else
299
300 /* Call the platform-specific actual file opener from mrapifil.h. */
301 retval = mfile_plt_open_read( filename, p_file );
302
303# endif /* MVFS_ENABLED */
304
305 return retval;

310void mfile_close( mfile_t* p_file ) {
311# ifdef MFILE_MMAP
312 munmap( bytes_ptr_h, bytes_sz );
313# else
314 /* maug_mfree( bytes_ptr_h ); */
315 switch( p_file->type ) {
316 case 0:
317 /* Do nothing silently. */
318 break;
319
320 case MFILE_CADDY_TYPE_FILE:
321 mfile_plt_close( p_file );
322 p_file->type = 0;
323 break;
324
325 case MFILE_CADDY_TYPE_MEM_BUFFER:
326 if( NULL != p_file->mem_buffer ) {
327 maug_munlock( p_file->h.mem, p_file->mem_buffer );
328 debug_printf( 1, "unlocked handle %p from file %p...",
329 p_file->h.mem, p_file );
330 p_file->type = 0;
331 }
332 break;
333
334 default:
335 error_printf( "unknown file type: %d", (p_file)->type );
336 break;
337 }
338# endif /* MFILE_MMAP */
339}
