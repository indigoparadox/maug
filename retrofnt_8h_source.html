61MERROR_RETVAL retrofont_load(
62 const char* font_name, MAUG_MHANDLE* p_font_h,
63 uint8_t glyph_h, uint16_t first_glyph, uint16_t glyphs_count
64) {
65 MERROR_RETVAL retval = MERROR_OK;
66 mfile_t font_file;
67 char line[RETROFONT_LINE_SZ];
68 struct RETROFONT* font = NULL;
69 char* line_bytes = NULL;
70 uint16_t glyph_idx = 0;
71 uint8_t* p_glyph = NULL;
72 size_t i = 0;
73 uint8_t glyph_w_bytes = 0;
74 uint8_t glyph_w = 0;
75
76 /* Try to separate the string into index:glyph bytes. */
77 #define retrofont_split_glyph_line( line, line_bytes ) \
78 line_bytes = strchr( line, ':' ); \
79 if( NULL == line_bytes ) { \
80 error_printf( "invalid line: %s", line ); \
81 retval = MERROR_PARSE; \
82 } \
83 line_bytes[0] = '\0'; \
84 line_bytes++;
85
86 retval = mfile_open_read( font_name, &font_file );
87 maug_cleanup_if_not_ok();
88
89 /* Figure out font width from file and alloc just enough. */
90 retval = mfile_read_line( &font_file, line, RETROFONT_LINE_SZ );
91 maug_cleanup_if_not_ok();
92 retrofont_split_glyph_line( line, line_bytes );
93 maug_cleanup_if_not_ok();
94 glyph_w_bytes = (strlen( line_bytes ) / glyph_h) / 2; /* 2 hex per byte */
95 debug_printf( 2, "glyph_w_bytes: %u", glyph_w_bytes );
96 glyph_w = glyph_w_bytes * 8;
97
98 debug_printf( 2, "glyph_w: %u, glyph_sz: %u",
99 glyph_w, glyph_h * glyph_w_bytes );
100
101 /* Alloc enough for each glyph, plus the size of the font header. */
102 *p_font_h = maug_malloc( 1,
103 sizeof( struct RETROFONT ) +
104 (glyph_h * glyph_w_bytes * (1 + glyphs_count)) );
105 maug_cleanup_if_null_alloc( MAUG_MHANDLE, *p_font_h );
106
107 debug_printf( 2, "allocated font %s: " SIZE_T_FMT " bytes",
108 font_name, (glyph_h * glyph_w_bytes * (1 + glyphs_count)) +
109 sizeof( struct RETROFONT ) );
110
111 maug_mlock( *p_font_h, font );
112 maug_cleanup_if_null_alloc( struct RETROFONT*, font );
113
114 /* Set initial font parameters. */
115 font->sz = sizeof( struct RETROFONT );
116 font->first_glyph = first_glyph;
117 font->glyph_w = glyph_w;
118 font->glyph_h = glyph_h;
119 font->glyph_sz = glyph_h * glyph_w_bytes;
120
121 while( mfile_has_bytes( &font_file ) ) {
122 retval = mfile_read_line( &font_file, line, RETROFONT_LINE_SZ );
123
124 retrofont_split_glyph_line( line, line_bytes );
125 if( MERROR_PARSE == retval ) {
126 retval = MERROR_OK;
127 continue;
128 }
129
130 /* Figure out the index of this glyph. */
131 /* TODO: Maug replacement for C99 crutch. */
132 glyph_idx = strtoul( line, NULL, 16 );
133 if( glyph_idx < first_glyph || glyph_idx > first_glyph + glyphs_count ) {
134 /* Skip glyph out of range. */
135 continue;
136 }
137
138 /* Find where to put the decoded glyph. */
139 p_glyph = retrofont_glyph_at( font, glyph_idx );
140
141 for( i = 0 ; font->glyph_h > i ; i++ ) {
142 switch( font->glyph_w ) {
143 case 8:
144 p_glyph[i] = 0;
145 p_glyph[i] |= maug_hctoi( line_bytes[i * 2] ) << 4;
146 p_glyph[i] |= maug_hctoi( line_bytes[(i * 2) + 1] );
147 break;
148
149 case 16:
150 /* TODO */
151 break;
152
153 case 32:
154 /* TODO */
155 break;
156
157 default:
158 error_printf( "invalid font width: %u", font->glyph_w );
159 retval = MERROR_PARSE;
160 goto cleanup;
161 }
162 }
163
164 if( glyph_idx == 'B' ) {
165 retrofont_dump_glyph( p_glyph, glyph_w, glyph_h );
166 }
167
168 font->glyphs_count++;
169
170 debug_printf( 1,
171 "%u %s (" SIZE_T_FMT " hbytes)", glyph_idx - first_glyph, line_bytes,
172 strlen( line_bytes ) );
173 }
174
175cleanup:
176
177 if( NULL != font ) {
178 maug_munlock( *p_font_h, font );
179 }
180
181 mfile_close( &font_file );
182
183 return retval;
184}

186void retrofont_blit_glyph(
187 struct RETROFLAT_BITMAP* target, RETROFLAT_COLOR color,
188 char c, struct RETROFONT* font, int x, int y, uint8_t flags
189) {
190 uint8_t* glyph = retrofont_glyph_at( font, c );
191 int16_t x_iter, y_iter;
192
193 for( y_iter = 0 ; font->glyph_h > y_iter ; y_iter++ ) {
194 for( x_iter = 0 ; font->glyph_w > x_iter ; x_iter++ ) {
195 if(
196 1 << (font->glyph_w - x_iter) ==
197 (glyph[y_iter] & (1 << (font->glyph_w - x_iter)))
198 ) {
199 retroflat_px( target, color, x + x_iter, y + y_iter, 0 );
200 }
201 }
202 }
