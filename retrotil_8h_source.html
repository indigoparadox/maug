459static void retrotile_parser_match_token(
460 const char* token, size_t token_sz, struct RETROTILE_PARSER* parser
461) {
462 size_t j = 1;
463
464 /* Figure out what the key is for. */
465 while( '\0' != gc_retrotile_mstate_tokens[j][0] ) {
466 if(
467 /* Make sure tokens match. */
468 maug_strlen( gc_retrotile_mstate_tokens[j] ) != token_sz ||
469 0 != strncmp(
470 token, gc_retrotile_mstate_tokens[j], token_sz
471 )
472 ) {
473 j++;
474 continue;
475
476 } else if(
477 /* This state can only be
478 * reached THROUGH that parent state. This allows us to have
479 * keys with the same name but different parents!
480 */
481 parser->mstate != gc_retrotile_mstate_parents[j]
482 ) {
483#ifdef RETROTILE_TRACE_TOKENS
484 debug_printf(
485 RETROTILE_TRACE_LVL,
486 "found token \"%s\" "
487#ifdef MPARSER_TRACE_NAMES
488 "but incorrect parent %s (%d) (needs %s (%d))!",
489#else
490 "but incorrect parent %d (needs %d)!",
491#endif /* MPARSER_TRACE_NAMES */
492 token,
493#ifdef MPARSER_TRACE_NAMES
494 retrotile_mstate_name( parser->mstate ),
495 parser->mstate,
496 retrotile_mstate_name( gc_retrotile_mstate_parents[j] ),
497 gc_retrotile_mstate_parents[j]
498#else
499 parser->mstate,
500 gc_retrotile_mstate_parents[j]
501#endif /* MPARSER_TRACE_NAMES */
502 );
503#endif /* !RETROTILE_TRACE_TOKENS */
504 j++;
505 continue;
506
507 } else if( parser->mode != gc_retrotile_mstate_modes[j] ) {
508#ifdef RETROTILE_TRACE_TOKENS
509 debug_printf(
510 RETROTILE_TRACE_LVL, "found token %s but incorrect mode %u!",
511 token,
512 gc_retrotile_mstate_modes[j] );
513#endif /* !RETROTILE_TRACE_TOKENS */
514 j++;
515 continue;
516
517 } else {
518 /* Found it! */
519#ifdef RETROTILE_TRACE_TOKENS
520 debug_printf(
521 RETROTILE_TRACE_LVL,
522 "found token \"%s\" "
523#ifdef MPARSER_TRACE_NAMES
524 "under correct parent %s (%d)!",
525#else
526 "under correct parent %d!",
527#endif /* MPARSER_TRACE_NAMES */
528 token,
529#ifdef MPARSER_TRACE_NAMES
530 retrotile_mstate_name( parser->mstate ),
531 parser->mstate
532#else
533 parser->mstate
534#endif /* MPARSER_TRACE_NAMES */
535 );
536#endif /* !RETROTILE_TRACE_TOKENS */
537
538 retrotile_parser_mstate( parser, j );
539 return;
540 }
541 }

546MERROR_RETVAL retrotile_parser_parse_tiledef_token(
547 const char* token, size_t token_sz, void* parser_arg
548) {
549 MERROR_RETVAL retval = MERROR_OK;
550 struct RETROTILE_TILE_DEF* tile_def = NULL;
551 struct RETROTILE_PARSER* parser = (struct RETROTILE_PARSER*)parser_arg;
552 size_t tileset_id_parsed = 0;
553
554 /* We don't lock the vector right away, since we might reallocate tile defs
555 * later on, below.
556 */
557
558 /* Try the custom parser. */
559 if(
560 NULL != parser->custom_token_cb &&
561 /* MERROR_PREEMPT is dealt with in cleanup. */
562 MERROR_OK != (retval = parser->custom_token_cb(
563 token, token_sz, parser_arg ))
564 ) {
565 goto cleanup;
566 }
567
568 if( MJSON_PSTATE_OBJECT_VAL != mjson_parser_pstate( &(parser->jparser) ) ) {
569 /* Not a token we have to worry about! */
570 retrotile_parser_match_token( token, token_sz, parser );
571 goto cleanup;
572 }
573
574 if( MTILESTATE_TILES_ID == parser->mstate ) {
575 retrotile_parser_mstate( parser, MTILESTATE_TILES );
576 if( 0 == parser->pass ) {
577 /* Parse tile ID. */
578 tileset_id_parsed = maug_atou32( token, token_sz, 10 );
579 if( tileset_id_parsed > parser->tileset_id_cur ) {
580 parser->tileset_id_cur = tileset_id_parsed;
581 debug_printf(
582 RETROTILE_TRACE_LVL,
583 "new highest tile ID: " SIZE_T_FMT, parser->tileset_id_cur );
584 }
585 } else {
586 assert( 0 < mdata_vector_ct( parser->p_tile_defs ) );
587 parser->tileset_id_cur = maug_atou32( token, token_sz, 10 );
588 debug_printf(
589 RETROTILE_TRACE_LVL,
590 "next tile ID: " SIZE_T_FMT, parser->tileset_id_cur );
591 }
592 retrotile_parser_mstate( parser, MTILESTATE_TILES );
593
594 } else if( MTILESTATE_TILES_IMAGE == parser->mstate ) {
595 if( 1 == parser->pass ) {
596 /* Need this pass 1 so tile_defs vector is allocated. */
597 mdata_vector_lock( parser->p_tile_defs );
598 tile_def = mdata_vector_get(
599 parser->p_tile_defs, parser->tileset_id_cur,
600 struct RETROTILE_TILE_DEF );
601 assert( NULL != tile_def );
602
603 /* Parse tile image. */
604 maug_strncpy( tile_def->image_path, token, RETROFLAT_ASSETS_PATH_MAX );
605
606 debug_printf(
607 RETROTILE_TRACE_LVL, "set tile ID " SIZE_T_FMT " to: %s",
608 parser->tileset_id_cur, tile_def->image_path );
609 }
610 retrotile_parser_mstate( parser, MTILESTATE_TILES );
611
612 } else if( MTILESTATE_TILES_CLASS == parser->mstate ) {
613 if( 1 == parser->pass ) {
614 /* Set the class from the pass 1 so the custom function can use
615 * it on pass 2 even if it's alphabetically out of order.
616 * (Tiled calls it "type" and puts it near the bottom!)
617 *
618 * We can't do this on pass 0 since tiles aren't allocated yet!
619 */
620 mdata_vector_lock( parser->p_tile_defs );
621 tile_def = mdata_vector_get(
622 parser->p_tile_defs, parser->tileset_id_cur,
623 struct RETROTILE_TILE_DEF );
624 assert( NULL != tile_def );
625 assert( 0 == tile_def->tile_class );
626
627 if( 0 == strncmp( "mobile", token, token_sz ) ) {
628 tile_def->tile_class = RETROTILE_CLASS_MOBILE;
629 debug_printf( RETROTILE_TRACE_LVL,
630 "set tile " SIZE_T_FMT " type: mobile (%u)",
631 parser->tileset_id_cur, tile_def->tile_class );
632 } else {
633 tile_def->tile_class = RETROTILE_CLASS_TILE;
634 debug_printf( RETROTILE_TRACE_LVL,
635 "set tile " SIZE_T_FMT " type: tile (%u)",
636 parser->tileset_id_cur, tile_def->tile_class );
637 }
638 }
639 retrotile_parser_mstate( parser, MTILESTATE_TILES );
640
641 } else if( MTILESTATE_TILES_PROP_NAME == parser->mstate ) {
642 debug_printf( RETROTILE_TRACE_LVL, "parsing property: %s", token );
643 maug_mzero( parser->last_prop_name, RETROTILE_PROP_NAME_SZ_MAX + 1 );
644 maug_strncpy( parser->last_prop_name, token, RETROTILE_PROP_NAME_SZ_MAX );
645 retrotile_parser_mstate( parser, MTILESTATE_TILES_PROP );
646
647 } else if( MTILESTATE_TILES_PROP_VAL == parser->mstate ) {
648 /* This should be handled in the custom token parser callback. */
649 retrotile_parser_mstate( parser, MTILESTATE_TILES_PROP );
650 }
651
652cleanup:
653
654 mdata_vector_unlock( parser->p_tile_defs );
655
656 if( MERROR_PREEMPT == retval ) {
657 /* Reset custom callback retval. */
658 retval = MERROR_OK;
659 }
660
661 return retval;
662}

666MERROR_RETVAL retrotile_parser_parse_token(
667 const char* token, size_t token_sz, void* parser_arg
668) {

672 retroflat_tile_t* tiles = NULL;
673
674 if( MJSON_PSTATE_LIST == mjson_parser_pstate( &(parser->jparser) ) ) {
675 /* We're parsing a list. What lists do we care about? */
676
677 /* Please note that this is for new tokens encountered inside of a list.
678 * For dealing with the closing of a list, see
679 * retrotile_json_close_list()!
680 */
681
682 if(
683 1 == parser->pass &&
684 MTILESTATE_LAYER_DATA == parser->mstate
685 ) {
686 /* This is a list of layer data... AKA tiles! */
687
688 /*
689 assert( NULL != *(parser->p_tilemap_h) );
690 maug_mlock( *(parser->p_tilemap_h), tilemap );
691 maug_cleanup_if_null_alloc( struct RETROTILE*, tilemap );
692 */
693
694 /*
695 debug_printf( RETROTILE_TRACE_LVL,
696 "selecting layer " SIZE_T_FMT "...",
697 parser->pass_layer_iter );
698 */
699 assert( NULL != parser->t );
700 tiles_layer = retrotile_get_layer_p(
701 parser->t, parser->pass_layer_iter );
702 assert( NULL != tiles_layer );
703
704 /* Apply the class parsed previously. */
705 tiles_layer->layer_class = parser->layer_class;
706
707 tiles = retrotile_get_tiles_p( tiles_layer );
708
709 /* Parse tilemap tile. */
710 /*
711 debug_printf( RETROTILE_TRACE_LVL,
712 "layer " SIZE_T_FMT " tile: " SIZE_T_FMT " (tiles: %p)",
713 parser->pass_layer_iter, parser->layer_tile_iter,
714 tiles );
715 */
716 assert( NULL != token );
717 assert( NULL != parser );
718 assert( NULL != tiles );
719
720 if(
721 parser->layer_tile_iter >=
722 parser->t->tiles_w * parser->t->tiles_h
723 ) {
724 /* Tile is outside of layer! */
725 error_printf(
726 "tile " SIZE_T_FMT " outside of layer tile buffer size "
727 SIZE_T_FMT "!",
728 parser->layer_tile_iter,
729 parser->t->tiles_w * parser->t->tiles_h );
730 retval = MERROR_OVERFLOW;
731 goto cleanup;
732 }
733
734 assert( 0 == tiles[parser->layer_tile_iter] );
735
736 tiles[parser->layer_tile_iter] = atoi( token );
737
738 }
739 parser->layer_tile_iter++;
740 goto cleanup;
741
742 } else if(
743 MJSON_PSTATE_OBJECT_VAL ==
744 mjson_parser_pstate( &(parser->jparser) )
745 ) {
746
747 /* Pay attention to the retrotile_parser_mstate() calls here. We don't
748 * have a stack of states, and we really only track a few we're
749 * interested in. Even if we don't process their data here, we still
750 * use retrotile_parser_mstate() to move the parser mstate back up
751 * to the parent mstate when we're done with it. This kind of "fakes" a
752 * stack (in a well-formed file, of course).
753 */
754
755 if( MTILESTATE_TILESETS_FGID == parser->mstate ) {
756 if( 1 == parser->pass ) {
757 parser->t->tileset_fgid = maug_atou32( token, token_sz, 10 );
758 debug_printf(
759 RETROTILE_TRACE_LVL, "tileset FGID set to: " SIZE_T_FMT,
760 parser->t->tileset_fgid );
761 }

834cleanup:
835

916 mparser_parse_token_cb token_cb, void* token_cb_data
917) {

920 struct RETROTILE_PARSER* parser = NULL;
921 char filename_path[RETROFLAT_PATH_MAX];
922 mfile_t buffer;
923 char c;
924 char* filename_ext = NULL;
925
926 /* Initialize parser. */
927 parser_h = maug_malloc( 1, sizeof( struct RETROTILE_PARSER ) );
928 maug_cleanup_if_null_alloc( MAUG_MHANDLE, parser_h );
929
930 maug_mlock( parser_h, parser );
931 maug_cleanup_if_null_alloc( struct RETROTILE_PARSER*, parser );
932 maug_mzero( parser, sizeof( struct RETROTILE_PARSER ) );
933
934 parser->tj_parse_cb = retrotile_parse_json_file;
935 parser->custom_token_cb = token_cb;
936 parser->custom_token_cb_data = token_cb_data;
937
938 maug_strncpy( parser->dirname, dirname, RETROFLAT_PATH_MAX );
939
940 /* Setup filename path. */
941 memset( filename_path, '\0', RETROFLAT_PATH_MAX );
942 /* TODO: Configurable path. */
943 maug_snprintf(
944 filename_path, RETROFLAT_PATH_MAX, "%s/%s", dirname, filename );
945
946 debug_printf( RETROTILE_TRACE_LVL, "opening %s...", filename_path );

948 retval = mfile_open_read( filename_path, &buffer );
949 maug_cleanup_if_not_ok();
950
951 /* Parse JSON and react to state. */
952 for( parser->pass = 0 ; 3 > parser->pass ; parser->pass++ ) {

954 parser->pass );
955
956 /* Reset tilemap parser. */

962 parser->wait_cb = wait_cb;
963 parser->wait_data = wait_data;
964 parser->jparser.base.wait_cb = wait_cb;
965 parser->jparser.base.wait_data = wait_data;
966
967 /* Figure out if we're parsing a .tmj or .tsj. */
968 filename_ext = maug_strrchr( filename, '.' );
969 if( NULL == filename_ext ) {
970 error_printf( "could not parse filename extension!" );
971 retval = MERROR_FILE;
972 goto cleanup;
973 }
974 if( 's' == filename_ext[2] ) {
975 debug_printf( RETROTILE_TRACE_LVL,
976 "(tile_defs pass %u)", parser->pass );
977 parser->mode = RETROTILE_PARSER_MODE_DEFS;
978 parser->jparser.token_parser = retrotile_parser_parse_tiledef_token;
979 parser->jparser.token_parser_arg = parser;
980 parser->jparser.close_list = retrotile_json_close_list;
981 parser->jparser.close_list_arg = parser;
982 parser->jparser.close_obj = retrotile_json_close_obj;
983 parser->jparser.close_obj_arg = parser;
984 /*
985 parser->jparser.base.close_val = retrotile_json_close_val;
986 parser->jparser.base.close_val_arg = parser;
987 */
988 parser->p_tile_defs = p_tile_defs;
989
990 assert( NULL != p_tile_defs );
991 if( 1 == parser->pass ) {
992 /* Allocate tile defs based on highest tile ID found on
993 * first pass.
994 */
995 assert( 0 < parser->tileset_id_cur );
996 debug_printf(
997 RETROTILE_TRACE_LVL, "allocating " SIZE_T_FMT " tile defs...",
998 parser->tileset_id_cur + 1 );
999 mdata_vector_fill(
1000 parser->p_tile_defs, parser->tileset_id_cur + 1,
1001 sizeof( struct RETROTILE_TILE_DEF ) );
1002 }
1003 } else {
1004 debug_printf( RETROTILE_TRACE_LVL, "(tilemap mode)" );
1005 parser->mode = RETROTILE_PARSER_MODE_MAP;
1006
1007 parser->jparser.close_list = retrotile_json_close_list;
1008 parser->jparser.close_list_arg = parser;
1009 parser->jparser.open_obj = retrotile_json_open_obj;
1010 parser->jparser.open_obj_arg = parser;
1011 parser->jparser.close_obj = retrotile_json_close_obj;
1012 parser->jparser.close_obj_arg = parser;
1013 parser->jparser.token_parser = retrotile_parser_parse_token;
1014 parser->jparser.token_parser_arg = parser;
1015 parser->p_tile_defs = p_tile_defs;
1016
1017 assert( NULL != p_tilemap_h );
1018 assert( NULL != p_tile_defs );
1019 if( 1 == parser->pass ) {
1020 /* Allocate tiles for the new layers. */
1021 retval = retrotile_alloc(
1022 p_tilemap_h, parser->tiles_w, parser->tiles_h,
1023 parser->pass_layer_iter, parser->tilemap_name );
1024 maug_cleanup_if_not_ok();
1025 maug_mlock( *p_tilemap_h, parser->t );
1026 }
1027 parser->pass_layer_iter = 0;
1028 }
1029
1030 while( mfile_has_bytes( &buffer ) ) {
1031 buffer.read_int( &buffer, (uint8_t*)&c, 1, 0 );

1033 if( MERROR_OK != retval ) {
1034 error_printf( "error parsing JSON!" );
1035 goto cleanup;
1036 }
1037 }
1038
1039 buffer.seek( &buffer, 0 );
1040
1041 filename_ext = maug_strrchr( filename, '.' );
1042 if( NULL == filename_ext ) {
1043 error_printf( "could not parse filename extension!" );
1044 retval = MERROR_FILE;
1045 goto cleanup;
1046 }
1047 if( 's' != filename_ext[2] ) {
1048 debug_printf( RETROTILE_TRACE_LVL,
1049 "pass %u found " SIZE_T_FMT " layers",
1050 parser->pass, parser->pass_layer_iter );
1051 }
1052 }
1053
1054 debug_printf(
1055 RETROTILE_TRACE_LVL, "finished parsing %s...", filename_path );
1056
1057cleanup:
1058
1059 if( NULL != parser ) {
1060 if( NULL != parser->t ) {
1061 maug_munlock( *p_tilemap_h, parser->t );
1062 }
1063 maug_munlock( parser_h, parser );
1064 }
1065
1066 if( NULL != parser_h ) {
1067 maug_mfree( parser_h );
1068 }
1069
1070 return retval;
1071}

1102
1103static void retrotile_gen_diamond_square_corners(
1104 int16_t corners_x[2][2], int16_t corners_y[2][2],
1105 retroflat_tile_t min_z, retroflat_tile_t max_z,
1106 uint32_t tuning, struct RETROTILE_DATA_DS* data_ds,
1107 struct RETROTILE_LAYER* layer, struct RETROTILE* t
1108) {
1109 int16_t iter_x = 0,
1110 iter_y = 0;
1111 retroflat_tile_t* tile_iter = NULL;
1112 retroflat_tile_t top_left_z = 0;
1113
1114 /* Generate missing corner data. Loop through X/Y coords stored in
1115 * corners_x/corners_y convenience arrays.
1116 */
1117 for( iter_y = 0 ; iter_y < 2 ; iter_y++ ) {
1118 for( iter_x = 0 ; iter_x < 2 ; iter_x++ ) {
1119
1120 /* Make sure corner X is in bounds. */
1121 corners_x[iter_x][iter_y] =
1122 (data_ds->sect_x - 1) + (iter_x * data_ds->sect_w);
1123 if( 0 > corners_x[iter_x][iter_y] ) {
1124 corners_x[iter_x][iter_y] += 1;
1125 }
1126
1127 /* Make sure corner Y is in bounds. */
1128 corners_y[iter_x][iter_y] =
1129 (data_ds->sect_y - 1) + (iter_y * data_ds->sect_h);
1130 if( 0 > corners_y[iter_x][iter_y] ) {
1131 corners_y[iter_x][iter_y] += 1;
1132 }
1133 }
1134 }
1135
1136 /* Should be handled by the check above. */
1137 assert( 0 <= corners_x[0][0] );
1138 assert( 0 <= corners_y[0][0] );
1139 assert( t->tiles_w > corners_x[0][0] );
1140 assert( t->tiles_h > corners_y[0][0] );
1141
1142 /* Grab the top-left Z-value to anchor generated corners to. */
1143 top_left_z = retrotile_get_tile(
1144 t, layer,
1145 corners_x[0][0],
1146 corners_y[0][0] );
1147
1148 if( 0 > top_left_z ) {
1149 retrotile_get_tile(
1150 t, layer,
1151 corners_x[0][0] >= 0 ? corners_x[0][0] : 0,
1152 corners_y[0][0] >= 0 ? corners_y[0][0] : 0 ) = max_z;
1153 top_left_z = max_z;
1154 }
1155
1156 /* Fill in missing corners. */
1157 for( iter_y = 0 ; iter_y < 2 ; iter_y++ ) {
1158 for( iter_x = 0 ; iter_x < 2 ; iter_x++ ) {
1159 /* Grab a pointer to the corner so we can modify it easily. */
1160 tile_iter = &(retrotile_get_tile(
1161 t, layer,
1162 corners_x[iter_x][iter_y],
1163 corners_y[iter_x][iter_y] ));
1164
1165 /* Check if corner is already filled in. */
1166 if( -1 != *tile_iter ) {
1167 debug_printf(
1168 RETROTILE_TRACE_LVL, "corner coord %d x %d present: %d",
1169 corners_x[iter_x][iter_y], corners_y[iter_x][iter_y],
1170 retrotile_get_tile(
1171 t, layer,
1172 corners_x[iter_x][iter_y],
1173 corners_y[iter_x][iter_y] ) );
1174 continue;
1175 }
1176
1177 /* Generate a new value for this corner. */
1178 *tile_iter = retrotile_gen_diamond_square_rand(
1179 min_z, max_z, tuning, top_left_z );
1180
1181 debug_printf( RETROTILE_TRACE_LVL,
1182 "missing corner coord %d x %d: %d",
1183 corners_x[iter_x][iter_y], corners_y[iter_x][iter_y],
1184 *tile_iter );
1185 }
1186 }
1187}

1226MERROR_RETVAL retrotile_gen_diamond_square_iter(
1227 struct RETROTILE* t, retroflat_tile_t min_z, retroflat_tile_t max_z,
1228 uint32_t tuning, size_t layer_idx, uint8_t flags, void* data,
1229 retrotile_ani_cb animation_cb, void* animation_cb_data
1230) {

1232 iter_y = 0,
1233 iter_depth = 0;
1234 int16_t corners_x[2][2];
1235 int16_t corners_y[2][2];
1236 int32_t avg = 0;
1237 /* size_t tile_idx = 0; */
1238 struct RETROTILE_DATA_DS data_ds_sub;
1239 MAUG_MHANDLE data_ds_h = NULL;
1240 struct RETROTILE_DATA_DS* data_ds = NULL;
1241 /* retroflat_tile_t* tiles = NULL; */
1242 MERROR_RETVAL retval = MERROR_OK;
1243 struct RETROTILE_LAYER* layer = NULL;
1244 retroflat_tile_t* tile_iter = NULL;
1245 uint8_t free_ds_data = 0;
1246
1247 /*
1248 maug_mlock( t->tiles, tiles );
1249 maug_cleanup_if_null_alloc( struct GRIDCITY_TILE*, tiles );
1250 */
1251
1252 #define _retrotile_ds_update_statistics( data, tile ) \
1253 /* Gather statistics. */ \
1254 if( (data)->highest_generated < (tile) && 32767 > (tile) ) { \
1255 (data)->highest_generated = (tile); \
1256 } \
1257 if( (data)->lowest_generated > (tile) && 0 < (tile) ) { \
1258 (data)->lowest_generated = (tile); \
1259 }
1260
1261 layer = retrotile_get_layer_p( t, layer_idx );
1262
1263 if(
1264 NULL == data ||
1265 RETROTILE_DS_FLAG_INIT_DATA == (RETROTILE_DS_FLAG_INIT_DATA & flags)
1266 ) {
1267 /* This must be the first call, so initialize or allocate a new struct.
1268 */
1269 if( NULL == data ) {
1270 /* An internal struct needs to be allocated before initialization. */
1271 data_ds_h = maug_malloc( 1, sizeof( struct RETROTILE_DATA_DS ) );
1272 maug_cleanup_if_null_alloc( MAUG_MHANDLE, data_ds_h );
1273 free_ds_data = 1;
1274 maug_mlock( data_ds_h, data_ds );
1275 maug_cleanup_if_null_alloc( struct RETROTILE_DATA_DS*, data_ds );
1276 } else {
1277 data_ds = (struct RETROTILE_DATA_DS*)data;
1278 }
1279
1280 /* Initialize passed tilemap while we're handling first call stuff. */
1281 memset( retrotile_get_tiles_p( layer ), -1,
1282 t->tiles_w * t->tiles_h * sizeof( retroflat_tile_t ) );
1283
1284 /* Initialize DS struct from tilemap properties. */
1285 maug_mzero( data_ds, sizeof( struct RETROTILE_DATA_DS ) );
1286 data_ds->sect_w = t->tiles_w;
1287 data_ds->sect_h = t->tiles_h;
1288 data_ds->sect_w_half = data_ds->sect_w >> 1;
1289 data_ds->sect_h_half = data_ds->sect_h >> 1;
1290 data_ds->lowest_generated = 32767;
1291
1292 /* Disable this flag for subsequent calls. */
1293 flags &= ~RETROTILE_DS_FLAG_INIT_DATA;
1294 } else {
1295 data_ds = (struct RETROTILE_DATA_DS*)data;
1296 }
1297 assert( NULL != data_ds );
1298
1299 /* Trivial case; end recursion. */
1300 if( 0 == data_ds->sect_w ) {
1301 debug_printf(
1302 RETROTILE_TRACE_LVL, "%d return: null sector", iter_depth );
1303 goto cleanup;
1304 }
1305
1306 if(
1307 data_ds->sect_x + data_ds->sect_w > t->tiles_w ||
1308 data_ds->sect_y + data_ds->sect_h > t->tiles_h
1309 ) {
1310 debug_printf(
1311 RETROTILE_TRACE_LVL, "%d return: overflow sector", iter_depth );
1312 goto cleanup;
1313 }
1314
1315 iter_depth = t->tiles_w / data_ds->sect_w;
1316
1317 /* Generate/grab corners before averaging them! */
1318 retrotile_gen_diamond_square_corners(
1319 corners_x, corners_y, min_z, max_z, tuning, data_ds, layer, t );
1320
1321 if( 2 == data_ds->sect_w || 2 == data_ds->sect_h ) {
1322 /* Nothing to average, this sector is just corners! */
1323 debug_printf(
1324 RETROTILE_TRACE_LVL,
1325 "%d return: reached innermost point", iter_depth );
1326 goto cleanup; /* Skip further descent regardless. */
1327 }
1328
1329 avg =
1330 retrotile_gen_diamond_square_avg( corners_x, corners_y, t, layer );
1331
1332 debug_printf( 1, "avg :%d", avg );
1333
1334 tile_iter = &(retrotile_get_tile(
1335 t, layer,
1336 data_ds->sect_x + data_ds->sect_w_half,
1337 data_ds->sect_y + data_ds->sect_h_half ));
1338 if( -1 != *tile_iter ) {
1339 debug_printf( RETROTILE_TRACE_LVL, "avg already present at %d x %d!",
1340 data_ds->sect_x + data_ds->sect_w_half,
1341 data_ds->sect_y + data_ds->sect_h_half );
1342 goto cleanup;
1343 }
1344 *tile_iter = avg;
1345 _retrotile_ds_update_statistics( data_ds, avg );
1346
1347 /* assert( 0 <= tiles[tile_idx].terrain );
1348
1349 maug_munlock( city->tiles, tiles );
1350 tiles = NULL; */
1351
1352 /* Recurse into subsectors. */
1353 for(
1354 iter_y = data_ds->sect_y ;
1355 iter_y < (data_ds->sect_y + data_ds->sect_h) ;
1356 iter_y++
1357 ) {
1358 for(
1359 iter_x = data_ds->sect_x ;
1360 iter_x < (data_ds->sect_x + data_ds->sect_w) ;
1361 iter_x++
1362 ) {
1363 data_ds_sub.sect_x = data_ds->sect_x + iter_x;
1364
1365 data_ds_sub.sect_y = data_ds->sect_y + iter_y;
1366
1367 data_ds_sub.sect_w = data_ds->sect_w_half;
1368 data_ds_sub.sect_h = data_ds->sect_h_half;
1369 data_ds_sub.sect_w_half = data_ds_sub.sect_w >> 1;
1370 data_ds_sub.sect_h_half = data_ds_sub.sect_h >> 1;
1371 data_ds_sub.lowest_generated = 32767;
1372 data_ds_sub.highest_generated = 0;
1373
1374 debug_printf(
1375 RETROTILE_TRACE_LVL, "%d: child sector at %d x %d, %d wide",
1376 iter_depth,
1377 data_ds_sub.sect_x, data_ds_sub.sect_y, data_ds_sub.sect_w );
1378
1379 retval = retrotile_gen_diamond_square_iter(
1380 t, min_z, max_z, tuning, layer_idx, flags, &data_ds_sub,
1381 animation_cb, animation_cb_data );
1382 maug_cleanup_if_not_ok();
1383
1384 _retrotile_ds_update_statistics(
1385 data_ds, data_ds_sub.highest_generated );
1386 _retrotile_ds_update_statistics(
1387 data_ds, data_ds_sub.lowest_generated );
1388 }
1389 }
1390
1391 if(
1392 data_ds->sect_w == t->tiles_w >> 1 &&
1393 NULL != animation_cb
1394 ) {
1395 retval = animation_cb( animation_cb_data, iter_y );
1396 maug_cleanup_if_not_ok();
1397 }
1398
1399 debug_printf(
1400 RETROTILE_TRACE_LVL, "%d return: all sectors complete", iter_depth );
1401
1402cleanup:
1403
1404 if( free_ds_data && NULL != data_ds ) {
1405 maug_munlock( data_ds_h, data_ds );
1406 }
1407
1408 if( free_ds_data && NULL != data_ds_h ) {
1409 maug_mfree( data_ds_h );
1410 }
1411
1412 return retval;

1417MERROR_RETVAL retrotile_gen_voronoi_iter(
1418 struct RETROTILE* t, retroflat_tile_t min_z, retroflat_tile_t max_z,
1419 uint32_t tuning, size_t layer_idx, uint8_t flags, void* data,
1420 retrotile_ani_cb animation_cb, void* animation_cb_data
1421) {
1422 size_t x = 0,
1423 y = 0;
1424 int16_t offset_x = 0,
1425 offset_y = 0,
1426 finished = 0;
1427 MERROR_RETVAL retval = MERROR_OK;
1428 struct RETROTILE_LAYER* layer = NULL;
1429 int16_t spb = RETROTILE_VORONOI_DEFAULT_SPB;
1430 int16_t drift = RETROTILE_VORONOI_DEFAULT_DRIFT;
1431 MAUG_MHANDLE temp_grid_h = (MAUG_MHANDLE)NULL;
1432 retroflat_tile_t* temp_grid = NULL;
1433 retroflat_tile_t* tiles = NULL;
1434 /* Only use 4 cardinal directions. */
1435 int8_t side_iter = 0;
1436
1437 layer = retrotile_get_layer_p( t, 0 );
1438
1439 tiles = retrotile_get_tiles_p( layer );
1440
1441 /* Initialize grid to empty. */
1442 memset( tiles, -1,
1443 t->tiles_w * t->tiles_h * sizeof( retroflat_tile_t ) );
1444
1445 /* Generate the initial sector starting points. */
1446 for( y = 0 ; t->tiles_w > y ; y += spb ) {
1447 for( x = 0 ; t->tiles_w > x ; x += spb ) {
1448 offset_x = x + ((drift * -1) + (rand() % drift));
1449 offset_y = y + ((drift * -1) + (rand() % drift));
1450
1451 /* Clamp sector offsets onto map borders. */
1452 if( 0 > offset_x ) {
1453 offset_x = 0;
1454 }
1455 if( offset_x >= t->tiles_w ) {
1456 offset_x = t->tiles_w - 1;
1457 }
1458 if( 0 > offset_y ) {
1459 offset_y = 0;
1460 }
1461 if( offset_y >= t->tiles_h ) {
1462 offset_y = t->tiles_h - 1;
1463 }
1464
1465 retrotile_get_tile( t, layer, offset_x, offset_y ) =
1466 min_z + (rand() % max_z);
1467 }
1468 }
1469
1470 temp_grid_h = maug_malloc(
1471 sizeof( retroflat_tile_t ), t->tiles_w * t->tiles_h );
1472 maug_cleanup_if_null_alloc( MAUG_MHANDLE, temp_grid_h );
1473
1474 maug_mlock( temp_grid_h, temp_grid );
1475 maug_cleanup_if_null_alloc( retroflat_tile_t*, temp_grid );
1476
1477 /* Grow the sector starting points. */
1478 while( !finished ) {
1479 if( NULL != animation_cb ) {
1480 retval = animation_cb( animation_cb_data, -1 );
1481 maug_cleanup_if_not_ok();
1482 }
1483
1484 /* Prepare sampling grid so we're working from unexpanded sections
1485 * below.
1486 */
1487 memcpy(
1488 temp_grid, tiles,
1489 sizeof( retroflat_tile_t ) * t->tiles_w * t->tiles_h );
1490
1491 /* Starting another pass, assume finished until proven otherwise. */
1492 finished = 1;
1493 for( y = 0 ; t->tiles_h > y ; y++ ) {
1494 for( x = 0 ; t->tiles_w > x ; x++ ) {
1495 if( -1 == retrotile_get_tile( t, layer, x, y ) ) {
1496 /* If there are still unfilled tiles, we're not finished
1497 * yet!
1498 */
1499 finished = 0;
1500
1501 /* Skip filled tile. */
1502 continue;
1503 }
1504
1505
1506 for( side_iter = 0 ; 4 > side_iter ; side_iter++ ) {
1507 debug_printf( RETROTILE_TRACE_LVL,
1508 SIZE_T_FMT " (%d), " SIZE_T_FMT " (%d) ("
1509 SIZE_T_FMT ", " SIZE_T_FMT ")",
1510 x,
1511 gc_retroflat_offsets4_x[side_iter],
1512 y,
1513 gc_retroflat_offsets4_y[side_iter],
1514 t->tiles_w, t->tiles_h );
1515
1516 /* Iterate through directions to expand. */
1517 /* TODO: Add tuning to select directional probability. */
1518 if(
1519 t->tiles_w > x + gc_retroflat_offsets4_x[side_iter] &&
1520 t->tiles_h > y + gc_retroflat_offsets4_y[side_iter] &&
1521 -1 == temp_grid[
1522 ((y + gc_retroflat_offsets4_y[side_iter]) *
1523 t->tiles_w) +
1524 (x + gc_retroflat_offsets4_x[side_iter])]
1525 ) {
1526 /* Copy center tile to this direction. */
1527 retrotile_get_tile( t, layer,
1528 x + gc_retroflat_offsets4_x[side_iter],
1529 y + gc_retroflat_offsets4_y[side_iter] ) =
1530 retrotile_get_tile( t, layer, x, y );
1531 break;
1532 }
1533 }
1534 }
1535 }
1536 }
1537
1538cleanup:
1539
1540 if( NULL != temp_grid ) {
1541 maug_munlock( temp_grid_h, temp_grid );
1542 }
1543
1544 if( NULL != temp_grid_h ) {
1545 maug_mfree( temp_grid_h );
1546 }
1547
1548 return retval;
1549}

1553MERROR_RETVAL retrotile_gen_smooth_iter(
1554 struct RETROTILE* t, retroflat_tile_t min_z, retroflat_tile_t max_z,
1555 uint32_t tuning, size_t layer_idx, uint8_t flags, void* data,
1556 retrotile_ani_cb animation_cb, void* animation_cb_data
1557) {
1558 MERROR_RETVAL retval = MERROR_OK;
1559 size_t x = 0,
1560 y = 0;
1561 int16_t side_iter = 0,
1562 sides_avail = 0,
1563 sides_sum = 0;
1564 /* Sides start from 12 on the clock (up). */
1565 struct RETROTILE_LAYER* layer = NULL;
1566
1567 assert( NULL != t );
1568 layer = retrotile_get_layer_p( t, layer_idx );
1569 assert( NULL != layer );
1570
1571 for( y = 0 ; t->tiles_h > y ; y++ ) {
1572 if( NULL != animation_cb ) {
1573 retval = animation_cb( animation_cb_data, y );
1574 maug_cleanup_if_not_ok();
1575 }
1576 for( x = 0 ; t->tiles_w > x ; x++ ) {
1577 /* Reset average. */
1578 sides_avail = 0;
1579 sides_sum = 0;
1580
1581 /* Grab values for available sides. */
1582 for( side_iter = 0 ; 8 > side_iter ; side_iter++ ) {
1583 if(
1584 t->tiles_w <= x + gc_retroflat_offsets8_x[side_iter] ||
1585 t->tiles_h <= y + gc_retroflat_offsets8_y[side_iter]
1586 ) {
1587 continue;
1588 }
1589
1590 sides_avail++;
1591 debug_printf(
1592 RETROTILE_TRACE_LVL,
1593 "si %d: x, y: " SIZE_T_FMT " (+%d), " SIZE_T_FMT
1594 " (+%d) idx: " SIZE_T_FMT,
1595 side_iter,
1596 x + gc_retroflat_offsets8_x[side_iter],
1597 gc_retroflat_offsets8_x[side_iter],
1598 y + gc_retroflat_offsets8_y[side_iter],
1599 gc_retroflat_offsets8_y[side_iter],
1600 ((y + gc_retroflat_offsets8_y[side_iter]) * t->tiles_w) +
1601 x + gc_retroflat_offsets8_x[side_iter] );
1602 sides_sum += retrotile_get_tile(
1603 t, layer,
1604 x + gc_retroflat_offsets8_x[side_iter],
1605 y + gc_retroflat_offsets8_y[side_iter] );
1606 }
1607
1608 retrotile_get_tile( t, layer, x, y ) = sides_sum / sides_avail;
1609 }
1610 }
1611
1612cleanup:
1613
1614 return retval;
1615}

1618
1619MERROR_RETVAL retrotile_gen_borders_iter(
1620 struct RETROTILE* t, retroflat_tile_t min_z, retroflat_tile_t max_z,
1621 uint32_t tuning, size_t layer_idx, uint8_t flags, void* data,
1622 retrotile_ani_cb animation_cb, void* animation_cb_data
1623) {
1624 MERROR_RETVAL retval = MERROR_OK;
1625 struct RETROTILE_DATA_BORDER* borders =
1626 (struct RETROTILE_DATA_BORDER*)data;
1627 size_t i = 0,
1628 x = 0,
1629 y = 0,
1630 x_plus_1 = 0,
1631 y_plus_1 = 0,
1632 side = 0;
1633 int16_t ctr_iter = 0,
1634 outside_iter = 0;
1635 struct RETROTILE_LAYER* layer = NULL;
1636
1637 assert( NULL != t );
1638 layer = retrotile_get_layer_p( t, layer_idx );
1639 assert( NULL != layer );
1640
1641 /* Reset tile counter for all defined borders. */
1642 for( i = 0 ; 0 <= borders[i].center ; i++ ) {
1643 borders[i].tiles_changed = 0;
1644 }
1645
1646 debug_printf( 1, "adding borders..." );
1647
1648 for( y = 0 ; t->tiles_h > y ; y++ ) {
1649 for( x = 0 ; t->tiles_w > x ; x++ ) {
1650 i = 0;
1651 while( 0 <= borders[i].center ) {
1652 /* Compare/grab current center tile. */
1653 ctr_iter = retrotile_get_tile( t, layer, x, y );
1654 debug_printf( RETROTILE_TRACE_LVL,
1655 "x: " SIZE_T_FMT ", y: " SIZE_T_FMT ", 0x%04x vs 0x%04x",
1656 x, y, ctr_iter, borders[i].center );
1657 if( ctr_iter != borders[i].center ) {
1658 i++;
1659 continue;
1660 }
1661
1662 debug_printf( RETROTILE_TRACE_LVL, "comparing sides..." );
1663
1664 /* Zeroth pass: look for stick-outs. */
1665 for( side = 0 ; 8 > side ; side += 2 ) {
1666 if(
1667 x + gc_retroflat_offsets8_x[side] > t->tiles_w ||
1668 y + gc_retroflat_offsets8_y[side] > t->tiles_h
1669 ) {
1670 /* Skip out-of-bounds. */
1671 continue;
1672 }
1673 /* Get the outside tile on this side. */
1674 outside_iter = retrotile_get_tile( t, layer,
1675 x + gc_retroflat_offsets8_x[side],
1676 y + gc_retroflat_offsets8_y[side] );
1677
1678 /* Get the outside tile next two clock-steps from this one.
1679 */
1680 if( side + 4 < 8 ) {
1681 x_plus_1 = x + gc_retroflat_offsets8_x[side + 4];
1682 y_plus_1 = y + gc_retroflat_offsets8_y[side + 4];
1683 } else {
1684 x_plus_1 = x + gc_retroflat_offsets8_x[side - 4];
1685 y_plus_1 = y + gc_retroflat_offsets8_y[side - 4];
1686 }
1687
1688 if(
1689 x_plus_1 < t->tiles_w && y_plus_1 < t->tiles_h &&
1690 outside_iter == borders[i].outside &&
1691 outside_iter == retrotile_get_tile( t, layer,
1692 x_plus_1, y_plus_1 )
1693 ) {
1694 /* This has the outside on two opposing sides, so just
1695 * erase it and use the outside. */
1696 retrotile_get_tile( t, layer, x, y ) =
1697 borders[i].outside;
1698 borders[i].tiles_changed++;
1699 goto tile_done;
1700 }
1701 }
1702
1703
1704 /* First pass: look for corners. */
1705 for( side = 0 ; 8 > side ; side += 2 ) {
1706 if(
1707 x + gc_retroflat_offsets8_x[side] > t->tiles_w ||
1708 y + gc_retroflat_offsets8_y[side] > t->tiles_h
1709 ) {
1710 /* Skip out-of-bounds. */
1711 continue;
1712 }
1713 /* Get the outside tile on this side. */
1714 outside_iter = retrotile_get_tile( t, layer,
1715 x + gc_retroflat_offsets8_x[side],
1716 y + gc_retroflat_offsets8_y[side] );
1717
1718 /* Get the outside tile next two clock-steps from this one.
1719 */
1720 if( side + 2 < 8 ) {
1721 x_plus_1 = x + gc_retroflat_offsets8_x[side + 2];
1722 y_plus_1 = y + gc_retroflat_offsets8_y[side + 2];
1723 } else {
1724 x_plus_1 = x + gc_retroflat_offsets8_x[0];
1725 y_plus_1 = y + gc_retroflat_offsets8_y[0];
1726 }
1727
1728 if(
1729 x_plus_1 < t->tiles_w && y_plus_1 < t->tiles_h &&
1730 outside_iter == borders[i].outside &&
1731 outside_iter == retrotile_get_tile( t, layer,
1732 x_plus_1, y_plus_1 )
1733 ) {
1734 /* This has the outside on two sides, so use a corner. */
1735 retrotile_get_tile( t, layer, x, y ) =
1736 borders[i].mod_to[side + 1 < 8 ? side + 1 : 0];
1737 borders[i].tiles_changed++;
1738 goto tile_done;
1739 }
1740 }
1741
1742 /* Second pass (if first pass fails): look for edges. */
1743 for( side = 0 ; 8 > side ; side += 2 ) {
1744 if(
1745 x + gc_retroflat_offsets8_x[side] > t->tiles_w ||
1746 y + gc_retroflat_offsets8_y[side] > t->tiles_h
1747 ) {
1748 /* Skip out-of-bounds. */
1749 continue;
1750 }
1751 /* Get the outside tile on this side. */
1752 outside_iter = retrotile_get_tile( t, layer,
1753 x + gc_retroflat_offsets8_x[side],
1754 y + gc_retroflat_offsets8_y[side] );
1755
1756 if( outside_iter == borders[i].outside ) {
1757 /* It only matches on this side. */
1758 debug_printf( RETROTILE_TRACE_LVL, "replacing..." );
1759 retrotile_get_tile( t, layer, x, y ) =
1760 borders[i].mod_to[side];
1761 borders[i].tiles_changed++;
1762 goto tile_done;
1763 }
1764 }
1765
1766tile_done:
1767 /* Tile replaced or not replaceable. */
1768 break;
1769 }
1770 }
1771 }
1772
1773 return retval;
1774}

1779 struct RETROTILE* tilemap, uint32_t layer_idx
1780) {

1797 layer_idx--;
1798 }

1881MERROR_RETVAL retrotile_topdown_draw(
1882 struct RETROFLAT_BITMAP* target,
1883 struct RETROTILE* t, struct MDATA_VECTOR* t_defs
1884) {
1885 int16_t x = 0,
1886 y = 0,
1887 x_tile = 0,
1888 y_tile = 0;
1889 retroflat_tile_t tile_id = 0;
1890 struct RETROTILE_LAYER* layer = NULL;
1891 struct RETROTILE_TILE_DEF* t_def = NULL;
1892 MERROR_RETVAL retval = MERROR_OK;
1893
1894 layer = retrotile_get_layer_p( t, 0 );
1895
1896 mdata_vector_lock( t_defs );
1897 /* TODO: Rework this so it uses viewport tile indexes and then calculates
1898 * screen pixel X/Y from those? For performance?
1899 */
1900 for(
1901 y = ((retroflat_viewport_world_y() >> 4) << 4) ;
1902 y < (int)retroflat_viewport_world_y() + (int)retroflat_viewport_screen_h()
1903 ; y += RETROFLAT_TILE_H
1904 ) {
1905 for(
1906 x = ((retroflat_viewport_world_x() >> 4) << 4) ;
1907 (int)retroflat_viewport_world_x() +
1908 (int)retroflat_viewport_screen_w() > x ;
1909 x += RETROFLAT_TILE_W
1910 ) {
1911 /* Limit to tiles that exist in the world. */
1912 if(
1913 -1 > x || -1 > y ||
1914 (int)retroflat_viewport_world_w() <= x ||
1915 (int)retroflat_viewport_world_h() <= y
1916 ) {
1917 continue;
1918 }
1919
1920 /* Divide by tile width (16), or shift by 2^4. */
1921 x_tile = x >> RETROFLAT_TILE_W_BITS;
1922 y_tile = y >> RETROFLAT_TILE_H_BITS;
1923
1924 tile_id = retrotile_get_tile( t, layer, x_tile, y_tile );
1925 t_def = mdata_vector_get(
1926 t_defs, tile_id - t->tileset_fgid, struct RETROTILE_TILE_DEF );
1927 assert( NULL != t_def );
1928
1929#ifndef RETROFLAT_NO_VIEWPORT_REFRESH
1930 /* Check tile refresh buffer. */
1931 retroflat_viewport_lock_refresh();
1932 if( !retroflat_viewport_tile_is_stale(
1933 x - retroflat_viewport_world_x(),
1934 y - retroflat_viewport_world_y(), tile_id
1935 ) ) {
1936 retroflat_viewport_unlock_refresh();
1937 continue;
1938 }
1939 retroflat_viewport_set_refresh( x, y, tile_id );
1940 retroflat_viewport_unlock_refresh();
1941#endif /* !RETROFLAT_NO_VIEWPORT_REFRESH */
1942
1943#ifdef RETROGXC_PRESENT
1944 retrogxc_blit_bitmap( target, t_def->image_cache_id,
1945 t_def->x, t_def->y,
1946 retroflat_viewport_screen_x( x ),
1947 retroflat_viewport_screen_y( y ),
1948 RETROFLAT_TILE_W, RETROFLAT_TILE_H,
1949 retroflat_instance_tile( tile_id ) );
1950#else
