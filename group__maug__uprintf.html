Value:
dest_idx += digits; \
while( 0 != num ) { \
/* Get the 1's place. */ \
rem = num % base; \
dest[--dest_idx] = (9 < rem) ? \
/* > 10, so it's a letter. */ \
(rem - 10) + 'a' : \
/* < 10, so it's a number. */ \
rem + '0'; \
/* Move the next place value into range. */ \
num /= base; \
digits_done++; \
} \
while( digits_done < digits ) { \
dest[--dest_idx] = '0'; \
digits_done++; \
}
