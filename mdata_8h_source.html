297ssize_t mdata_strpool_find(
298 struct MDATA_STRPOOL* strpool, const char* str, size_t str_sz
299) {
300 MERROR_RETVAL retval = MERROR_OK;
301 ssize_t i = 0;
302 char* strpool_p = NULL;
303 size_t* p_str_iter_sz = NULL;
304
305 if( NULL == strpool->str_h ) {
306 error_printf( "strpool not allocated!" );
307 i = -1;
308 goto cleanup;
309 }
310
311 maug_mlock( strpool->str_h, strpool_p );
312
313 while( i < strpool->str_sz ) {
314 p_str_iter_sz = (size_t*)&(strpool_p[i]);
315 if(
316 0 == strncmp( &(strpool_p[i + sizeof( size_t )]), str, str_sz + 1 )
317 ) {
318 /* String found. Advance past the size before returning. */
319 i += sizeof( size_t );
320 debug_printf( MDATA_TRACE_LVL,
321 "found strpool_idx: " SIZE_T_FMT " (" SIZE_T_FMT " bytes): \"%s\"",
322 i, *p_str_iter_sz, &(strpool_p[i]) );
323
324 goto cleanup;
325 } else {
326 debug_printf( MDATA_TRACE_LVL,
327 "skipping strpool_idx: " SIZE_T_FMT " (" SIZE_T_FMT
328 " bytes): \"%s\"",
329 i + sizeof( size_t ), *p_str_iter_sz,
330 &(strpool_p[i + sizeof( size_t )]) );
331 }
332 i += *p_str_iter_sz;
333 }
334
335 /* String not found. */
336 i = -1;
337
338cleanup:
339
340 if( MERROR_OK != retval ) {
341 i = retval * -1;
342 }
343
344 if( NULL != strpool_p ) {
345 maug_munlock( strpool->str_h, strpool_p );
346 }
347
348 return i;
349}

353MAUG_MHANDLE mdata_strpool_extract( struct MDATA_STRPOOL* s, size_t i ) {
354 MERROR_RETVAL retval = MERROR_OK;
355 char* strpool = NULL;
356 MAUG_MHANDLE out_h = NULL;
357 size_t out_sz = 0;
358 char* out_tmp = NULL;
359
360 mdata_strpool_lock( s, strpool );
361
362 out_sz = maug_strlen( &(strpool[i]) );
363 out_h = maug_malloc( out_sz + 1, 1 );
364 maug_cleanup_if_null_alloc( MAUG_MHANDLE, out_h );
365
366 maug_mlock( out_h, out_tmp );
367 maug_cleanup_if_null_lock( char*, out_tmp );
368
369 maug_mzero( out_tmp, out_sz + 1 );
370 maug_strncpy( out_tmp, &(strpool[i]), out_sz );
371
372cleanup:
373
374 if( NULL != out_tmp ) {
375 maug_munlock( out_h, out_tmp );
376 }
377
378 if( MERROR_OK != retval && NULL != out_h ) {
379 maug_mfree( out_h );
380 }
381
382 if( NULL != strpool ) {
383 mdata_strpool_unlock( s, strpool );
384 }
385
386 return out_h;
387}

391ssize_t mdata_strpool_append(
392 struct MDATA_STRPOOL* strpool, const char* str, size_t str_sz
393) {
394 ssize_t idx_p_out = 0;
395 char* strpool_p = NULL;
396 MERROR_RETVAL retval = MERROR_OK;
397 size_t* p_str_sz = NULL;
398 size_t padding = 0;
399 size_t alloc_sz = 0;
400
401 if( 0 < strpool->str_sz ) {
402 /* Search the str_stable for an identical string and return that index.
403 */
404 idx_p_out = mdata_strpool_find( strpool, str, str_sz );
405 if( -1 != idx_p_out ) {
406 /* Found, or error returned. */
407 goto cleanup;
408 }
409 }
410
411 assert( 0 < str_sz );
412
413 /* Pad out allocated space so size_t is always aligned. */
414 padding = sizeof( size_t ) - ((str_sz + 1 /* NULL */) % sizeof( size_t ));
415 alloc_sz = sizeof( size_t ) + str_sz + 1 /* NULL */ + padding;
416 assert( 0 == alloc_sz % sizeof( size_t ) );
417
418 debug_printf( MDATA_TRACE_LVL,
419 "adding size_t (" SIZE_T_FMT " bytes) + string %s (" SIZE_T_FMT
420 " bytes) + 1 NULL + " SIZE_T_FMT " bytes padding to strpool...",
421 sizeof( size_t ), str, str_sz, padding );
422
423 retval = mdata_strpool_alloc( strpool, alloc_sz );
424 maug_cleanup_if_not_ok();
425
426 maug_mlock( strpool->str_h, strpool_p );
427 maug_cleanup_if_null_alloc( char*, strpool_p );
428
429 debug_printf( MDATA_TRACE_LVL,
430 "strpool (" SIZE_T_FMT " bytes) locked to: %p",
431 strpool->str_sz, strpool_p );
432
433 /* Add this string at the end of the string table. */
434 maug_strncpy( &(strpool_p[strpool->str_sz + sizeof( size_t )]), str, str_sz );
435 strpool_p[strpool->str_sz + sizeof( size_t ) + str_sz] = '\0';
436
437 /* Add the size of the string to the strpool. */
438 assert( 0 == strpool->str_sz % sizeof( size_t ) );
439 p_str_sz = (size_t*)&(strpool_p[strpool->str_sz]);
440 *p_str_sz = alloc_sz;
441
442 idx_p_out = strpool->str_sz + sizeof( size_t );
443
444 debug_printf(
445 MDATA_TRACE_LVL, "set strpool_idx: " SIZE_T_FMT ": \"%s\"",
446 strpool->str_sz, &(strpool_p[idx_p_out]) );
447
448 /* Set the string table cursor to the next available spot. */
449 strpool->str_sz += alloc_sz;
450
451cleanup:
452
453 if( MERROR_OK != retval ) {
454 idx_p_out = retval * -1;
455 }
456
457 if( NULL != strpool_p ) {
458 maug_munlock( strpool->str_h, strpool_p );
459 }
460
461 return idx_p_out;
462}

504ssize_t mdata_vector_append(
505 struct MDATA_VECTOR* v, const void* item, size_t item_sz
506) {
507 MERROR_RETVAL retval = MERROR_OK;
508 ssize_t idx_out = -1;
509
510 if( 0 < v->item_sz && item_sz != v->item_sz ) {
511 error_printf( "attempting to add item of " SIZE_T_FMT " bytes to vector, "
512 "but vector is already sized for " SIZE_T_FMT "-byte items!",
513 item_sz, v->item_sz );
514 retval = MERROR_OVERFLOW;
515 goto cleanup;
516 }
517
518 mdata_vector_alloc( v, item_sz, MDATA_VECTOR_INIT_SZ );
519
520 /* Lock the vector to work in it a bit. */
521 mdata_vector_lock( v );
522
523 idx_out = v->ct;
524
525 if( NULL != item ) {
526 /* Copy provided item. */
527 debug_printf(
528 MDATA_TRACE_LVL, "inserting into vector at index: " SIZE_T_FMT,
529 idx_out );
530
531 memcpy( _mdata_vector_item_ptr( v, idx_out ), item, item_sz );
532 }
533
534 v->ct++;
535
536cleanup:
537
538 if( MERROR_OK != retval ) {
539 error_printf( "error adding to vector: %d", retval );
540 idx_out = retval * -1;
541 assert( 0 > idx_out );
542 }
543
544 mdata_vector_unlock( v );
545
546 return idx_out;
547}

551MERROR_RETVAL mdata_vector_remove( struct MDATA_VECTOR* v, size_t idx ) {
552 MERROR_RETVAL retval = MERROR_OK;
553 size_t i = 0;
554
555 if( NULL != v->data_bytes ) {
556 error_printf( "vector cannot be resized while locked!" );
557 retval = MERROR_ALLOC;
558 goto cleanup;
559 }
560
561 if( v->ct <= idx ) {
562 error_printf( "index out of range!" );
563 retval = MERROR_OVERFLOW;
564 goto cleanup;
565 }
566
567 debug_printf( MDATA_TRACE_LVL, "removing vector item: " SIZE_T_FMT, idx );
568
569 assert( 0 < v->item_sz );
570
571 mdata_vector_lock( v );
572
573 for( i = idx ; v->ct > i + 1 ; i++ ) {
574 debug_printf( MDATA_TRACE_LVL,
575 "shifting " SIZE_T_FMT "-byte vector item " SIZE_T_FMT " up by 1...",
576 v->item_sz, i );
577 memcpy(
578 &(v->data_bytes[i * v->item_sz]),
579 &(v->data_bytes[(i + 1) * v->item_sz]),
580 v->item_sz );
581 }
582
583 v->ct--;
584
585cleanup:
586
587 mdata_vector_unlock( v );
588
589 return retval;
590}

651MERROR_RETVAL mdata_vector_alloc(
652 struct MDATA_VECTOR* v, size_t item_sz, size_t item_ct_init
653) {
654 MERROR_RETVAL retval = MERROR_OK;
655 MAUG_MHANDLE data_h_new = NULL;
656 size_t new_ct = item_ct_init,
657 new_bytes_start = 0,
658 new_bytes_sz = 0;
659
660 if( NULL != v->data_bytes ) {
661 error_printf( "vector cannot be resized while locked!" );
662 retval = MERROR_ALLOC;
663 goto cleanup;
664 }
665
666 /* Make sure there are free nodes. */
667 if( (MAUG_MHANDLE)NULL == v->data_h ) {
668 assert( 0 == v->ct_max );
669
670 v->ct_max = item_ct_init;
671 v->ct_step = MDATA_VECTOR_INIT_STEP_SZ;
672 debug_printf(
673 MDATA_TRACE_LVL,
674 "creating " SIZE_T_FMT " vector of " SIZE_T_FMT "-byte nodes...",
675 v->ct_max, item_sz );
676 v->data_h = maug_malloc( v->ct_max, item_sz );
677 v->item_sz = item_sz;
678 maug_cleanup_if_null_alloc( MAUG_MHANDLE, v->data_h );
679
680 /* Zero out the new space. */
681 mdata_vector_lock( v );
682 maug_mzero( v->data_bytes, v->ct_max * item_sz );
683 mdata_vector_unlock( v );
684
685 } else if( v->ct_max <= v->ct + 1 || v->ct_max <= item_ct_init ) {
686 assert( item_sz == v->item_sz );
687
688 /* Use ct * 2 or ct_init... whichever is larger! */
689 if( item_ct_init < v->ct_max + v->ct_step ) {
690 assert( v->ct_max + v->ct_step > v->ct_max );
691 new_ct = v->ct_max + v->ct_step;
692 }
693
694 /* Perform the resize. */
695 debug_printf(
696 MDATA_TRACE_LVL, "enlarging vector to " SIZE_T_FMT "...",
697 new_ct );
698 maug_mrealloc_test( data_h_new, v->data_h, new_ct, item_sz );
699
700 /* Zero out the new space. */
701 new_bytes_start = v->ct_max * v->item_sz;
702 assert( new_bytes_start >= v->ct_max );
703 new_bytes_sz = (new_ct * v->item_sz) - new_bytes_start;
704 assert( new_bytes_sz >= v->item_sz );
705 mdata_vector_lock( v );
706 maug_mzero( &(v->data_bytes[new_bytes_start]), new_bytes_sz );
707 mdata_vector_unlock( v );
708
709 v->ct_max = new_ct;
710 }
711
712cleanup:
713
714 return retval;
715}
